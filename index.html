function init() { const urlParams = new URLSearchParams(window.location.search); // 確保這裡抓的是 'gem' const gemId = urlParams.get('gem'); let saved = JSON.parse(localStorage.getItem('zpd_9gems') || '[]'); if (gemId && gems[gemId]) { // 如果網址有 ?gem=1，強制進入開機流程並跳出寶石 document.getElementById('boot-screen').style.display = 'flex'; if (!saved.includes(gemId)) { saved.push(gemId); localStorage.setItem('zpd_9gems', JSON.stringify(saved)); } setTimeout(() => { showAlert(gemId, true); }, 2500); } else { // 如果網址沒有參數，2秒後自動關閉開機畫面顯示紀錄簿 setTimeout(() => { const boot = document.getElementById('boot-screen'); if(boot) boot.style.display = 'none'; }, 2000); } updateUI(saved); }
